## 원형 큐의 Enqueue와 Dequeue 구현
```cpp
#include <stdio.h>
#define MAX 5

typedef struct {
	int items[MAX];
	int rear;
	int front;
} CircularQueue;

void initQueue(CircularQueue* q) {
	q->rear = -1;
	q->front = -1;
}

int isEmpty(CircularQueue* q) {
	return q->front == -1;
}

int isFull(CircularQueue* q) {
	return (q->rear + 1) % MAX == q->front;
}

void enqueue(CircularQueue* q, int num) {
	if (isFull(q)) {
		printf("isfull\n");
		return;
	}
	if (isEmpty(q)) {
		q->front = 0;
	}
	q->rear = (q->rear + 1) % MAX;
	q->items[q->rear] = num;
}

int dequeue(CircularQueue* q) {
	if (isEmpty(q)) {
		printf("isEmpty\n");
		return -1;
	}
	//
	int val = q->items[q->front];
	if (q->front == q->rear) {
		initQueue(q);
	}
	else { q->front = (q->front + 1) % MAX; }
	return val;
}

int main() {
	CircularQueue q;
	initQueue(&q);

	dequeue(&q);
	enqueue(&q, 1);
	enqueue(&q, 2);
	enqueue(&q, 3);
	enqueue(&q, 4);
	enqueue(&q, 5); // full
	enqueue(&q, 6); // err

	for (int i = 0; i < MAX; i++) {
		printf("%d ", q.items[i]);
	}

	printf("\n");

	printf("%d\n",dequeue(&q));
	printf("%d\n",dequeue(&q));

	enqueue(&q, 7);
	printf("%d\n",dequeue(&q));
	
	enqueue(&q, 6);
	printf("%d\n", dequeue(&q));

	for (int i = 0; i < MAX; i++) {
		printf("%d ", q.items[i]);
	}

	printf("\n");
	printf("%d\n", dequeue(&q));

	return 0;
}
```
